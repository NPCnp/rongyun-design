<script>
    // ---------- 基础数据 ----------
    const STORAGE_KEY = 'diy_orders';
    const ORDER_NO_KEY = 'last_order_no';
    let fabricCanvas = null;
    let currentMockup = 'postcard';
    let historyStack = [];
    let historyIndex = -1;
    let currentCategory = "三坊七巷";

    const themeData = {
        "三坊七巷": ["🏯 古厝飞檐", "🏮 石板巷", "🪷 马鞍墙", "🖼️ 爱心树"],
        "福州特色": ["🎪 福", "🌼 茉莉花", "🍡 鱼丸", "🏮 红墙"],
        "上下杭": ["⛵ 上下杭", "🌃 夜景"],
        "西禅寺": ["🏯 西禅寺"],
        "地标建筑": ["🏯 白塔", "🏯 乌塔", "🏛️ 镇海楼"]
    };
    const themeColor = {
        "三坊七巷": "#b35f2d",
        "福州特色": "#8fbc8f",
        "上下杭": "#5b7c5c",
        "西禅寺": "#b8865b",
        "地标建筑": "#4a6fa5"
    };
    const canvasSizes = {
        postcard: { width: 500, height: 333 },
        tote: { width: 375, height: 500 }
    };
    const presetPhrases = ['有福之州', '平安喜乐', '长乐未央', '岁岁平安', '家和万事兴', '福到家门'];

    function showToast(msg) {
        const toast = document.getElementById('toast');
        toast.innerText = msg;
        toast.classList.add('show');
        setTimeout(() => toast.classList.remove('show'), 2000);
    }

    // 初始化画布
    function initCanvas() {
        const canvasEl = document.getElementById('designCanvas');
        const size = canvasSizes[currentMockup];
        canvasEl.width = size.width;
        canvasEl.height = size.height;
        if (fabricCanvas) fabricCanvas.dispose();
        fabricCanvas = new fabric.Canvas('designCanvas', {
            width: size.width,
            height: size.height,
            selection: true,
            preserveObjectStacking: true,
            enableTouchScrolling: false,
            uniScaleTransform: true,
            centeredScaling: false
        });
        // 设置控制点便于手指操作
        fabric.Object.prototype.set({
            cornerSize: 24,
            cornerTouchSize: 48,
            cornerColor: '#B84A3C',
            borderColor: '#C9A87C',
            transparentCorners: false,
            selectable: true,
            hasControls: true,
            hasBorders: true
        });
        if (currentMockup === 'postcard') {
            fabricCanvas.setBackgroundColor('#ffffff', fabricCanvas.renderAll.bind(fabricCanvas));
        } else {
            fabricCanvas.setBackgroundColor('#F5F0E6', fabricCanvas.renderAll.bind(fabricCanvas));
        }
        // 添加默认文字示例
        const defaultText = new fabric.Text('福州·三坊七巷', {
            left: 50,
            top: 50,
            fontSize: 20,
            fill: '#b45f2b',
            fontFamily: 'Noto Serif SC'
        });
        fabricCanvas.add(defaultText);
        fabricCanvas.renderAll();
        saveHistory();
        updatePreviewThumb();
    }

    function saveHistory() {
        if (!fabricCanvas) return;
        const json = JSON.stringify(fabricCanvas.toJSON());
        historyStack = historyStack.slice(0, historyIndex + 1);
        historyStack.push(json);
        historyIndex++;
        if (historyStack.length > 20) historyStack.shift();
    }

    function undo() {
        if (historyIndex > 0) {
            historyIndex--;
            fabricCanvas.loadFromJSON(historyStack[historyIndex], () => {
                fabricCanvas.renderAll();
                updatePreviewThumb();
                showToast('撤销');
            });
        } else {
            showToast('没有可撤销的操作');
        }
    }

    function zoomIn() {
        const active = fabricCanvas.getActiveObject();
        if (!active) { showToast('请先选中一个元素'); return; }
        let scale = (active.scaleX || 1) + 0.1;
        if (scale > 3) scale = 3;
        active.set('scaleX', scale);
        active.set('scaleY', scale);
        fabricCanvas.renderAll();
        saveHistory();
        updatePreviewThumb();
    }

    function zoomOut() {
        const active = fabricCanvas.getActiveObject();
        if (!active) { showToast('请先选中一个元素'); return; }
        let scale = (active.scaleX || 1) - 0.1;
        if (scale < 0.3) scale = 0.3;
        active.set('scaleX', scale);
        active.set('scaleY', scale);
        fabricCanvas.renderAll();
        saveHistory();
        updatePreviewThumb();
    }

    function deleteSelected() {
        const active = fabricCanvas.getActiveObject();
        if (active) {
            fabricCanvas.remove(active);
            fabricCanvas.renderAll();
            saveHistory();
            updatePreviewThumb();
            showToast('已删除');
        } else {
            showToast('请先选中一个元素');
        }
    }

    function updatePreviewThumb() {
        const thumbImg = document.getElementById('previewThumbImg');
        if (!thumbImg) return;
        thumbImg.src = fabricCanvas.toDataURL('image/png');
    }

    function switchMode(mode) {
        if (mode === currentMockup) return;
        const oldJson = JSON.stringify(fabricCanvas.toJSON());
        currentMockup = mode;
        const oldSize = canvasSizes[mode === 'postcard' ? 'tote' : 'postcard'];
        const newSize = canvasSizes[mode];
        const parsed = JSON.parse(oldJson);
        if (parsed.objects) {
            parsed.objects.forEach(obj => {
                const relX = obj.left / oldSize.width;
                const relY = obj.top / oldSize.height;
                obj.left = relX * newSize.width;
                obj.top = relY * newSize.height;
                obj.scaleX = obj.scaleX || 1;
                obj.scaleY = obj.scaleY || 1;
            });
        }
        const newJson = JSON.stringify(parsed);
        initCanvas();
        fabricCanvas.loadFromJSON(newJson, () => {
            fabricCanvas.renderAll();
            saveHistory();
            updatePreviewThumb();
        });
        showToast(`已切换至${mode === 'postcard' ? '明信片' : '帆布袋'}模式`);
    }

    function addPattern(name) {
        const parts = name.split(' ');
        const label = parts.slice(1).join(' ') || name;
        const centerX = fabricCanvas.width / 2;
        const centerY = fabricCanvas.height / 2;
        const textObj = new fabric.Text(label, {
            left: centerX,
            top: centerY,
            fontSize: 28,
            fontFamily: 'Noto Serif SC',
            fill: themeColor[currentCategory] || '#b45f2b',
            originX: 'center',
            originY: 'center'
        });
        fabricCanvas.add(textObj);
        fabricCanvas.setActiveObject(textObj);
        fabricCanvas.renderAll();
        saveHistory();
        updatePreviewThumb();
        showToast(`已添加: ${label}`);
    }

    function addUserText(customText) {
        let content = customText || document.getElementById('userTextInput').value.trim();
        if (!content) content = '福州·有福';
        const fontFamily = document.getElementById('fontFamilySelect').value;
        const fontSize = parseInt(document.getElementById('fontSizeSelect').value);
        const color = document.getElementById('textColorPicker').value;
        const centerX = fabricCanvas.width / 2;
        const centerY = fabricCanvas.height / 2;
        const textObj = new fabric.Text(content, {
            left: centerX,
            top: centerY,
            fontSize: fontSize,
            fontFamily: fontFamily,
            fill: color,
            originX: 'center',
            originY: 'center'
        });
        fabricCanvas.add(textObj);
        fabricCanvas.setActiveObject(textObj);
        fabricCanvas.renderAll();
        saveHistory();
        updatePreviewThumb();
        showToast(`已添加文字: ${content}`);
        if (!customText) document.getElementById('userTextInput').value = '';
    }

    function openOrderModal() {
        const orderNo = (parseInt(localStorage.getItem(ORDER_NO_KEY) || '0') + 1);
        document.getElementById('orderNo').value = orderNo;
        document.getElementById('orderName').value = '';
        document.getElementById('orderPhone').value = '';
        document.getElementById('orderModal').classList.add('active');
    }

    function closeOrderModal() {
        document.getElementById('orderModal').classList.remove('active');
    }

    function submitOrder() {
        const orderNo = (parseInt(localStorage.getItem(ORDER_NO_KEY) || '0') + 1);
        const name = document.getElementById('orderName').value.trim();
        const phone = document.getElementById('orderPhone').value.trim();
        if (!name) { showToast('请输入姓名'); return; }
        if (!phone || !/^1[3-9]\d{9}$/.test(phone)) { showToast('请输入正确的手机号'); return; }
        localStorage.setItem(ORDER_NO_KEY, orderNo);
        const design = fabricCanvas.toDataURL('image/png');
        const pickupCode = Math.floor(100000 + Math.random() * 900000);
        const orders = JSON.parse(localStorage.getItem(STORAGE_KEY) || '[]');
        orders.push({
            orderNo, name, phone, design,
            time: new Date().toLocaleString(),
            timestamp: Date.now(),
            status: '待制作',
            pickupCode
        });
        localStorage.setItem(STORAGE_KEY, JSON.stringify(orders));
        showToast(`订单已提交！取件码：${pickupCode}`);
        closeOrderModal();
    }

    function showPreview() {
        const fullImg = document.getElementById('previewFullImg');
        fullImg.src = fabricCanvas.toDataURL('image/png');
        document.getElementById('previewModal').classList.add('active');
    }
    function closePreview() {
        document.getElementById('previewModal').classList.remove('active');
    }

    function renderCategories() {
        const container = document.getElementById('categoryTabs');
        container.innerHTML = '';
        Object.keys(themeData).forEach(cat => {
            const btn = document.createElement('div');
            btn.className = `cat-btn ${cat === currentCategory ? 'active' : ''}`;
            btn.innerText = cat;
            btn.onclick = () => {
                currentCategory = cat;
                renderCategories();
                renderPatterns();
            };
            container.appendChild(btn);
        });
    }

    function renderPatterns() {
        const grid = document.getElementById('patternGrid');
        grid.innerHTML = '';
        const patterns = themeData[currentCategory];
        patterns.forEach(name => {
            const div = document.createElement('div');
            div.className = 'pattern-item';
            const emoji = name.split(' ')[0];
            const label = name.split(' ').slice(1).join(' ') || name;
            div.innerHTML = `<div class="pattern-emoji">${emoji}</div><div class="pattern-label">${label}</div>`;
            div.onclick = () => addPattern(name);
            grid.appendChild(div);
        });
    }

    function initPresetPhrases() {
        const container = document.getElementById('presetPhrases');
        presetPhrases.forEach(p => {
            const span = document.createElement('span');
            span.className = 'preset-phrase';
            span.innerText = p;
            span.onclick = () => addUserText(p);
            container.appendChild(span);
        });
    }

    function bindEvents() {
        document.getElementById('undoBtn').onclick = undo;
        document.getElementById('zoomInBtn').onclick = zoomIn;
        document.getElementById('zoomOutBtn').onclick = zoomOut;
        document.getElementById('deleteBtn').onclick = deleteSelected;
        document.getElementById('orderBtn').onclick = openOrderModal;
        document.getElementById('addTextBtn').onclick = () => addUserText();
        document.getElementById('closeOrderBtn').onclick = closeOrderModal;
        document.getElementById('submitOrderBtn').onclick = submitOrder;
        document.getElementById('previewThumb').onclick = showPreview;
        document.getElementById('closePreviewBtn').onclick = closePreview;
        document.querySelectorAll('.mode-btn').forEach(btn => {
            btn.onclick = () => {
                document.querySelectorAll('.mode-btn').forEach(b => b.classList.remove('active'));
                btn.classList.add('active');
                switchMode(btn.dataset.mode);
            };
        });
    }

    window.onload = () => {
        renderCategories();
        renderPatterns();
        initPresetPhrases();
        initCanvas();
        bindEvents();
        window.addEventListener('resize', () => setTimeout(() => updatePreviewThumb(), 100));
        showToast('长按元素可拖拽移动，点击按钮缩放');
    };
</script>
